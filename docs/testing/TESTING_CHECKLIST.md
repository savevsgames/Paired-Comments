## üìù Test Writing Guidelines

### Unit Test Template
```typescript
import { describe, it } from 'mocha';
import { expect } from 'chai';

describe('ModuleName', () => {
  describe('methodName', () => {
    it('should handle happy path', () => {
      // Arrange
      const input = 'test';

      // Act
      const result = methodName(input);

      // Assert
      expect(result).to.equal('expected');
    });

    it('should handle edge case', () => {
      // Test edge case
    });

    it('should throw on invalid input', () => {
      // Test error handling
    });
  });
});
```

### E2E Test Template
```typescript
import * as vscode from 'vscode';
import { suite, test } from 'mocha';
import { expect } from 'chai';

suite('Feature Name', () => {
  test('should perform user workflow', async () => {
    // Arrange - Set up VS Code environment
    const doc = await vscode.workspace.openTextDocument({ language: 'javascript' });

    // Act - Perform actions
    await vscode.commands.executeCommand('pairedComments.addComment');

    // Assert - Verify results
    expect(result).to.exist;
  });
});
```
